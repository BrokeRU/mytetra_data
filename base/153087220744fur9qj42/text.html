# arbitrary number of seconds

s = 13420

# hours

hours = s // 3600

# remaining seconds

s = s - (hours * 3600)

# minutes

minutes = s // 60

# remaining seconds

seconds = s - (minutes * 60)

# total time

print '%s:%s:%s' % (hours, minutes, seconds)

# result: 3:43:40
