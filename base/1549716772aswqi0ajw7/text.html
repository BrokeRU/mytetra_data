Чтобы заставить VSCode понимать нестандартные импорты, например:


import 'utils/str';


, который на самом деле должен быть зарезолвлен в src/utils/str.js, нужно создать файл со следующим содержимым:


// jsconfig.json

{

"compilerOptions": {

"target": "ES6",

"module": "commonjs",

"allowSyntheticDefaultImports": true,

"baseUrl": "./",

"paths": {

"components/*": ["src/components/*"],

"utils/*": ["src/utils/*"],

"redux/*": ["src/redux/*"],

"routes/*": ["src/routes/*"],

"api/*": ["src/api/*"]

}

}

}
