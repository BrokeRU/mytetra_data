from datetime import datetime


dt = datetime.now()


# 1) str.format

strg = '{:%B %d, %Y}'.format(dt)

print(strg) # July 22, 2017


# 2) datetime.strftime

strg = dt.strftime('%B %d, %Y')

print(strg) # July 22, 2017


# 3) f-strings in python >= 3.6

strg = f'{dt:%B %d, %Y}'

print(strg) # July 22, 2017
