Чтобы установить рабочую директорию для отладчика в VSCode нужно добавить пункт cwd в конфигурацию запуска:


{

"version": "0.2.0",

"configurations": [

{

"name": "Python: Current File (Integrated Terminal)",

"type": "python",

"request": "launch",

"program": "${file}",

"console": "integratedTerminal",

"cwd": "${fileDirname}"

},


//... other settings, but I modified the "Current File" setting above ...

}
