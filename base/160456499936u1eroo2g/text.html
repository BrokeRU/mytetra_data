Константы -- это именованные значения, которые создаются на этапе компиляции. Значение константы нельзя поменять.


Простое объявление константы:


const pi float64 = 3.1415


Объявление в блоке:


const (

a int = 45

b float32 = 3.3

)


Если в блоке не указать значение одной из констант, то она примет значение предыдущей константы:


package main


import (

"fmt"

)


const(

A int = 45

B

C float32 = 3.3

D

)


func main() {

fmt.Println(A, B, C, D) // Вывод: 45 45 3.3 3.3

}
